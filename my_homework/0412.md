package day16.quiz;

import javax.swing.JOptionPane;

class Student {
	private String name;
	private int kr, en, ma;
	private double avg;
	private boolean pass;
	private char grade;
	
	// 1. 생성자 (여러분 마음대로 여러 개 만들기) 
	
	// 2. getters (형식에 맞게)
	
	// 3. setters (재료를 넣으면 검열을 거쳐 올바른 데이터만 들어감)
	//  0) name : 이름
	//  1) kr, en, ma : 0 점 이상 100점 이하만 저장 가능, 그 외 0점 
	//  2) avg : (인자값 받지 않고, kr, en, ma만 가지고 계산되도록)
	//  3) pass : 60점 이상이면 true
	//  4) grade : ABCDF 중 1개로
	
	
	
	
	
	public void setName(String name) {
		if (name == null || name.isEmpty() || name.length() < 0) {
			return;         

		}

		this.name = name;
	}
	
	public void setKr(int kr) {
		if(kr < 0 || kr > 100) {
			return;
		}
		this.kr = kr;
	}
	
	public void setEn(int en) {
		if(en < 0 || en > 100) {
			return;
		}
		this.en = en;
	}
	
	public void setMa(int ma) {
		if(ma < 0 || ma > 100) {
			return;
		}
		this.ma = ma;
	}
	
	
	
	public void setAvg() {
		
		double avg = (kr + en + ma) / 3.0;
		if(avg < 0) {
			return;
		}
		this.avg = avg;
	}
	
	public void setGrade() {
		if(avg >= 90) {
			grade = 'A';
		}
		else if(avg >=80) {
			grade = 'B';
		}
		else if(avg >=70) {
			grade = 'C';
		}
		else if(avg >=60) {
			grade = 'D';
		}
		else 
			grade = 'F';
		
	}
	
	public void setPass() {
		this.pass = this.avg >= 60 ? true : false;
	}
	
	public String getName() {
		return name;
	}
	public int getKr() {
		return kr;
	}
	public int getEn() {
		return en;
	}
	public int getMa() {
		return ma;
	}
	public double getAvg() {
		return avg;
	}
	
	public char getGrade() {
		return grade;
	}
	
	public boolean isPass() {
		return pass;
	}
	
	

	
	
	
	
	
	
}




public class Quiz01 {

public static void main(String[] args) {
		
		Student s = new Student();
		
		
		s.setName("박연재");
		s.setKr(40);
		s.setEn(40);
		s.setMa(40);
		
	
		
		
		s.setAvg();
		s.setPass();
		s.setGrade();
		
		System.out.println("이름 : " + s.getName());
		System.out.println("국어 점수 : " + s.getKr());
		System.out.println("영어 점수 : " + s.getEn());
		System.out.println("수학 점수 : " + s.getMa());
		System.out.println("세 과목의 평균점수 : " + s.getAvg());
		System.out.println("성적 : " + s.getGrade());
		System.out.println("합격? : " + s.isPass());
		
		
		
		
		
		
		
		
		
		
		
	}
}
